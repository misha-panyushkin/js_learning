Переменные

Формируем слова

var day ;
var month ;
var year ;

var day , month , year ;

day = 10; month = 07; year = 2013;

Объявляем и присваиваем.

var day = 10,
    month = 07,
    year = 2013;

Слабая типизация.

var day;
day = 10;
day = "10";
day = "среда.";
day = "третий день от воскресенья!";

day = true;  // логическе значение

Операции над переменными

var year = 2013;

Перенесёмся в будущее.
year + 10;   // Результат операции никуда не сохранен!

Сохраняем результат.
var future = year + 10;   // future = 2023

Слово не воробей!

var future = 2023;

Здесь всё понятно, полетели дальше.
var far_future = Future + 10;

  Результат операции - ошибка!

Назвался с маленькой буквы, носи такое имя.
var far_future = future + 10;

Моя первая функция

function SayMyName () {
       // Частная собственность!
}

При обращении к имени SayMyName, будет выведено:
function SayMyName () {...}

Складываем слова в предложения:

function SayMyName (name) {
      var str = name;
      str = "Hello" + name + "!";
      alert (str);
       // Вызываем функцию внутри другой функции
}

Циклы

В отличие от условий, можно повторять:


for(var i = 0; i < 100; i++) {
      ...
}

В обратном порядке:


for(var i = 100; i > 0; i--) {
      ...
}

Атака массивов


"цветок", "черный дипломат", "очки", "плащ"
"леон", "матильда", "стетфилд", "цветок"
1, 2, 3, 4

Всё это напоминает очередь

Беспорядочный список:
"цветок", "черный дипломат", "очки", "плащ"

Упорядочный список:
"подъём", "упрожнения на пресс", "в магазин за молоком", "завтрак"

Окаймляем в скобки:
["подъём", "упрожнения на пресс", "в магазин за молоком", "завтрак"];

Записываем в переменную:
var morning = ["подъём", "упрожнения на пресс", "в магазин за молоком", "завтрак"];

Утренний распорядок:
var morning = ["подъём", "упрожнения на пресс", "в магазин за молоком", "завтрак"];


  • morning[0] => "подъём"
  • morning[1] => "упрожнения на пресс"
  • morning[2] => "в магазин за молоком"
  • morning[3] => "завтрак"
